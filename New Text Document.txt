                    </button>

                          {faq.answer}
                        </p>
                      </div>
                    )}
                  </div>
                ))}
              </div>

              {/* Add to Cart Button */}
              <div className="text-center mt-12 lg:mt-12 xl:mt-12 2xl:mt-16">
                <button

                >
                  ADD TO CART
                </button>

          </div>
